STEP 1 - CREATE MEASURES USING THE DAX FORMULAS BELOW: 

1.  CurrentRate = CALCULATE( AVERAGE('Retention'[RetentionRate]),(Retention[RetentionED]))

 

2. Start = CALCULATE(AVERAGE('Retention'[StartP]),Retention[RetentionED])

3. New = CALCULATE(AVERAGE('Retention'[NewP]),Retention[RetentionED])

4. Final = CALCULATE( AVERAGE('Retention'[EndP]),Retention[RetentionED])

5. Drop = [Start]+[New] -[Final]


6. Drop% = DIVIDE([Drop],[Start]+[New],0)


7. KPIArrow = (SWITCH(true(),Retention[CurrentRate]>= .5,UNICHAR(9650),[CurrentRate]<= .49,UNICHAR(9660)))


8. KPIArrow2 = (SWITCH(true(),Retention[Drop%]>= .5,UNICHAR(9650),[Drop%]<= .49,UNICHAR(9660)))


9. KPIColors = (SWITCH(true(),Retention[CurrentRate]>= .5,"#299954",[CurrentRate]<= .49,"#e83815"))


10. KPIColors2 = (SWITCH(true(),Retention[Drop%]>= .5,"#299954",[Drop%]<= .49,"#e83815"))


11. WeeklyDifdropout = var MaxWEEK = MAX(Retention[WkDiff]) var MinWeek= MAX(Retention[WkDiff])-1

RETURN if (CALCULATE(SUM(Retention[dropout]),Retention[WkDiff]=MaxWEEK)<>0,CALCULATE(SUM(Retention[dropout]),

Retention[WkDiff]=MaxWEEK) - CALCULATE(SUM(Retention[dropout]),Retention[WkDiff]=MinWeek))

12. .TotalDrop = If ([.WeeklyDifdropout] <>0 , [.WeeklyDifdropout])
